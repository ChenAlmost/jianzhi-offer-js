## 题目描述
给一个链表，若其中包含环，请找出该链表的环的入口结点，否则，输出 null。

## 答案

```javascript
/*function ListNode(x){
    this.val = x;
    this.next = null;
}*/
function EntryNodeOfLoop(pHead) {
  var current = pHead;
  while (current.next != null) {
    current.flag = true;
    if (current.next.flag == true) {
      return current.next;
    }
    current = current.next;
  }
  return null;
}
module.exports = {
  EntryNodeOfLoop: EntryNodeOfLoop,
};
```
